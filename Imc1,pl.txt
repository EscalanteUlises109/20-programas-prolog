%% Programa para calcular el Índice de Masa Corporal
%% Programación Lógica y Funcional

imc(Peso, Altura, IMC) :-
    IMC is Peso / (Altura * Altura).

categoria_imc(IMC, Categoria) :-
    IMC < 18.5,
    Categoria = 'Bajo peso'.
categoria_imc(IMC, Categoria) :-
    IMC >= 18.5,
    IMC < 25,
    Categoria = 'Normal'.
categoria_imc(IMC, Categoria) :-
    IMC >= 25,
    IMC < 30,
    Categoria = 'Sobrepeso'.
categoria_imc(IMC, Categoria) :-
    IMC >= 30,
    Categoria = 'Obesidad'.
